Time & Space Complexity

Common Time Complexities

Notation Name Example Operations
O(1) Constant Array access, Hash table insertion/lookup
O(log n) Logarithmic Binary search, Balanced BST operations
O(n) Linear Linear search, Traversing an array
O(n log n) Linearithmic Efficient sorting (merge, heap, quick)
O(n) Quadratic Nested loops, Bubble/Insertion/Selection sort
O(2) Exponential Recursive Fibonacci, Power set
O(n!) Factorial Permutations, Traveling Salesman (brute force)

Space Complexity Considerations

  • In-place algorithms use O(1) extra space
  • Recursive algorithms use extra space on the call stack (proportional to depth)
  • Consider input size vs. auxiliary space used

Memory Management

// Stack allocation (automatic)
int x = 10;  // Automatically freed when out of scope

// Heap allocation (dynamic)
int* ptr = new int;  // Allocate single integer
*ptr = 10;
delete ptr;  // Must free manually

// Dynamic array
int* arr = new int[10];  // Allocate array of 10 integers
// ... use the array ...
delete[] arr;  // Free array memory

// Smart pointers (modern C++)
#include <memory>

// Unique pointer (exclusive ownership)
std::unique_ptr<int> uptr = std::make_unique<int>(10);
// No need to delete, automatically freed when uptr goes out of scope

// Shared pointer (shared ownership)
std::shared_ptr<int> sptr1 = std::make_shared<int>(20);
std::shared_ptr<int> sptr2 = sptr1;  // Reference count = 2
// Freed when all shared_ptrs go out of scope

Linked Lists

// Node class
class Node {
public:
    int data;
    Node* next;
    
    Node(int val) : data(val), next(nullptr) {}
};

// Singly linked list implementation
class LinkedList {
private:
    Node* head;
    
public:
    LinkedList() : head(nullptr) {}
    
    // Insert at beginning
    void insertAtBeginning(int val) {
        Node* newNode = new Node(val);
        newNode->next = head;
        head = newNode;
    }
    
    // Insert at end
    void insertAtEnd(int val) {
        Node* newNode = new Node(val);
        
        if (head == nullptr) {
            head = newNode;
            return;
        }
        
        Node* temp = head;
        while (temp->next != nullptr) {
            temp = temp->next;
        }
        
        temp->next = newNode;
    }
    
    // Delete a node with given value
    void deleteNode(int val) {
        if (head == nullptr) return;
        
        // If head needs to be removed
        if (head->data == val) {
            Node* temp = head;
            head = head->next;
            delete temp;
            return;
        }
        
        // Find the node to be deleted
        Node* curr = head;
        Node* prev = nullptr;
        
        while (curr != nullptr && curr->data != val) {
            prev = curr;
            curr = curr->next;
        }
        
        // If value not found
        if (curr == nullptr) return;
        
        // Unlink and delete
        prev->next = curr->next;
        delete curr;
    }
    
    // Display the list
    void display() {
        Node* temp = head;
        while (temp != nullptr) {
            cout << temp->data << " -> ";
            temp = temp->next;
        }
        cout << "nullptr" << endl;
    }
    
    // Destructor to free memory
    ~LinkedList() {
        Node* current = head;
        Node* next;
        
        while (current != nullptr) {
            next = current->next;
            delete current;
            current = next;
        }
        
        head = nullptr;
    }
};

Time Complexity:

  • Access: O(n)
  • Search: O(n)
  • Insertion at beginning: O(1)
  • Insertion at end: O(n) (O(1) with tail pointer)
  • Deletion: O(n)

class Stack {
private:
    struct Node {
        int data;
        Node* next;
        Node(int val) : data(val), next(nullptr) {}
    };
    
    Node* top;
    
public:
    Stack() : top(nullptr) {}
    
    void push(int val) {
        Node* newNode = new Node(val);
        newNode->next = top;
        top = newNode;
    }
    
    bool isEmpty() {
        return top == nullptr;
    }
    
    int peek() {
        if (isEmpty()) {
            throw runtime_error("Stack is empty");
        }
        return top->data;
    }
    
    void pop() {
        if (isEmpty()) {
            throw runtime_error("Stack is empty");
        }
        
        Node* temp = top;
        top = top->next;
        delete temp;
    }
    
    ~Stack() {
        while (!isEmpty()) {
            pop();
        }
    }
};

Time Complexity:

  • Push: O(1)
  • Pop: O(1)
  • Top: O(1)
  • Search: O(n)

// Custom queue implementation using linked list
class Queue {
private:
    struct Node {
        int data;
        Node* next;
        Node(int val) : data(val), next(nullptr) {}
    };
    
    Node* front;
    Node* rear;
    
public:
    Queue() : front(nullptr), rear(nullptr) {}
    
    void enqueue(int val) {
        Node* newNode = new Node(val);
        
        if (rear == nullptr) {
            front = rear = newNode;
            return;
        }
        
        rear->next = newNode;
        rear = newNode;
    }
    
    void dequeue() {
        if (front == nullptr) {
            throw runtime_error("Queue is empty");
        }
        
        Node* temp = front;
        front = front->next;
        
        if (front == nullptr) {
            rear = nullptr;
        }
        
        delete temp;
    }
    
    int peek() {
        if (front == nullptr) {
            throw runtime_error("Queue is empty");
        }
        return front->data;
    }
    
    bool isEmpty() {
        return front == nullptr;
    }
    
    ~Queue() {
        while (!isEmpty()) {
            dequeue();
        }
    }
};

Time Complexity:

  • Enqueue: O(1)
  • Dequeue: O(1)
  • Front/Back: O(1)
  • Search: O(n)

Trees

Binary Search Tree (BST)

class BST {
private:
    struct Node {
        int data;
        Node* left;
        Node* right;
        
        Node(int val) : data(val), left(nullptr), right(nullptr) {}
    };
    
    Node* root;
    
    // Helper methods
    Node* insertHelper(Node* node, int val) {
        if (node == nullptr) {
            return new Node(val);
        }
        
        if (val < node->data) {
            node->left = insertHelper(node->left, val);
        } else if (val > node->data) {
            node->right = insertHelper(node->right, val);
        }
        
        return node;
    }
    
    bool searchHelper(Node* node, int val) {
        if (node == nullptr) {
            return false;
        }
        
        if (node->data == val) {
            return true;
        }
        
        if (val < node->data) {
            return searchHelper(node->left, val);
        } else {
            return searchHelper(node->right, val);
        }
    }
    
    void inorderHelper(Node* node) {
        if (node == nullptr) return;
        
        inorderHelper(node->left);
        cout << node->data << " ";
        inorderHelper(node->right);
    }
    
    Node* findMin(Node* node) {
        while (node->left != nullptr) {
            node = node->left;
        }
        return node;
    }
    
    Node* deleteHelper(Node* node, int val) {
        if (node == nullptr) return nullptr;
        
        if (val < node->data) {
            node->left = deleteHelper(node->left, val);
        } else if (val > node->data) {
            node->right = deleteHelper(node->right, val);
        } else {
            // Case 1: No children
            if (node->left == nullptr && node->right == nullptr) {
                delete node;
                return nullptr;
            }
            // Case 2: One child
            else if (node->left == nullptr) {
                Node* temp = node->right;
                delete node;
                return temp;
            } else if (node->right == nullptr) {
                Node* temp = node->left;
                delete node;
                return temp;
            }
            // Case 3: Two children
            else {
                Node* temp = findMin(node->right);
                node->data = temp->data;
                node->right = deleteHelper(node->right, temp->data);
            }
        }
        
        return node;
    }
    
    void destroyTree(Node* node) {
        if (node == nullptr) return;
        
        destroyTree(node->left);
        destroyTree(node->right);
        delete node;
    }
    
public:
    BST() : root(nullptr) {}
    
    void insert(int val) {
        root = insertHelper(root, val);
    }
    
    bool search(int val) {
        return searchHelper(root, val);
    }
    
    void remove(int val) {
        root = deleteHelper(root, val);
    }
    
    void inorder() {
        inorderHelper(root);
        cout << endl;
    }
    
    ~BST() {
        destroyTree(root);
    }
};

// Inorder Traversal (Left, Root, Right)

// Preorder Traversal (Root, Left, Right)

    if (root == nullptr) return;

    preorder(root->right);

// Postorder Traversal (Left, Right, Root)
void postorder(Node* root) {
    if (root == nullptr) return;
    postorder(root->left);
    postorder(root->right);
    cout << root->data << " ";
}

// Level Order Traversal (BFS)
void levelOrder(Node* root) {
    if (root == nullptr) return;
    
    queue<Node*> q;
    q.push(root);
    
    while (!q.empty()) {
        Node* current = q.front();
        q.pop();
        
        cout << current->data << " ";
        
        if (current->left) q.push(current->left);
        if (current->right) q.push(current->right);
    }
}

BST Time Complexity:

  • Search: O(log n) average, O(n) worst
  • Insertion: O(log n) average, O(n) worst
  • Deletion: O(log n) average, O(n) worst

Custom Heap Implementation:

class MinHeap {
private:
    vector<int> heap;
    
    int parent(int i) { return (i - 1) / 2; }
    int leftChild(int i) { return 2 * i + 1; }
    int rightChild(int i) { return 2 * i + 2; }
    
    void heapifyUp(int i) {
        while (i > 0 && heap[parent(i)] > heap[i]) {
            swap(heap[i], heap[parent(i)]);
            i = parent(i);
        }
    }
    
    void heapifyDown(int i) {
        int minIndex = i;
        int left = leftChild(i);
        int right = rightChild(i);
        int size = heap.size();
        
        if (left < size && heap[left] < heap[minIndex]) {
            minIndex = left;
        }
        
        if (right < size && heap[right] < heap[minIndex]) {
            minIndex = right;
        }
        
        if (i != minIndex) {
            swap(heap[i], heap[minIndex]);
            heapifyDown(minIndex);
        }
    }
    
public:
    void push(int val) {
        heap.push_back(val);
        heapifyUp(heap.size() - 1);
    }
    
    int top() {
        if (heap.empty()) {
            throw runtime_error("Heap is empty");
        }
        return heap[0];
    }
    
    void pop() {
        if (heap.empty()) {
            throw runtime_error("Heap is empty");
        }
        
        heap[0] = heap.back();
        heap.pop_back();
        
        if (!heap.empty()) {
            heapifyDown(0);
        }
    }
    
    bool empty() {
        return heap.empty();
    }
    
    int size() {
        return heap.size();
    }
};

Heap Time Complexity:

  • Insert: O(log n)
  • Remove top element: O(log n)
  • Get top element: O(1)
  • Heapify an array: O(n)

        // adj[v].push_back({u, weight});
    }
    
    void removeEdge(int u, int v) {
        for (auto it = adj[u].begin(); it != adj[u].end(); ++it) {
            if (it->first == v) {
                adj[u].erase(it);
                break;
            }
        }
        
        // For undirected graph:
        // Similar code for adj[v]
    }
    
    bool isEdge(int u, int v) {
        for (auto& edge : adj[u]) {
            if (edge.first == v) return true;
        }
        return false;
    }
    
    int getWeight(int u, int v) {
        for (auto& edge : adj[u]) {
            if (edge.first == v) return edge.second;
        }
        return 0;  // No edge
    }

        vector<int> neighbors;
        for (auto& edge : adj[u]) {
            neighbors.push_back(edge.first);
        }
        return neighbors;
    }
    
    int getVertexCount() {
        return V;
    }
};

Graph Traversals

// Breadth-First Search (BFS)
void BFS(GraphList& graph, int start) {
    int V = graph.getVertexCount();
    vector<bool> visited(V, false);
    queue<int> q;
    
    visited[start] = true;
    q.push(start);
    
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        cout << u << " ";
        
        vector<int> neighbors = graph.getNeighbors(u);
        for (int v : neighbors) {
            if (!visited[v]) {
                visited[v] = true;
                q.push(v);
            }
        }
    }
}

// Depth-First Search (DFS) - Recursive
void DFSUtil(GraphList& graph, int u, vector<bool>& visited) {
    visited[u] = true;
    cout << u << " ";
    
    vector<int> neighbors = graph.getNeighbors(u);
    for (int v : neighbors) {
        if (!visited[v]) {
            DFSUtil(graph, v, visited);
        }
    }
}

void DFS(GraphList& graph, int start) {
    int V = graph.getVertexCount();
    vector<bool> visited(V, false);
    
    DFSUtil(graph, start, visited);
}

// DFS - Iterative
void DFSIterative(GraphList& graph, int start) {
    int V = graph.getVertexCount();
    vector<bool> visited(V, false);
    stack<int> s;
    
    s.push(start);
    
    while (!s.empty()) {
        int u = s.top();
        s.pop();
        
        if (!visited[u]) {
            visited[u] = true;
            cout << u << " ";
            
            vector<int> neighbors = graph.getNeighbors(u);
            for (int i = neighbors.size() - 1; i >= 0; i--) {
                int v = neighbors[i];
                if (!visited[v]) {
                    s.push(v);
                }
            }
        }
    }
}

Graph Time Complexity:

  • Adjacency Matrix:
    • Storage: O(V)
    • Add/Remove Edge: O(1)
    • Check Edge: O(1)
    • Find all neighbors: O(V)
  • Adjacency List:
    • Storage: O(V + E)
    • Add Edge: O(1)
    • Remove Edge: O(E)
    • Check Edge: O(E)
    • Find all neighbors: O(E)
  • BFS/DFS: O(V + E)

Bubble Sort

void bubbleSort(vector<int>& arr) {
    int n = arr.size();
    bool swapped;
    
    for (int i = 0; i < n - 1; i++) {
        swapped = false;
        
        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                swap(arr[j], arr[j + 1]);
                swapped = true;
            }
        }
        
        // If no swapping occurred in this pass, array is sorted
        if (!swapped) break;
    }
}

Time Complexity:

  • Best Case: O(n) when already sorted
  • Average Case: O(n)
  • Worst Case: O(n)
  • Space Complexity: O(1)

Selection Sort

void selectionSort(vector<int>& arr) {
    int n = arr.size();
    
    for (int i = 0; i < n - 1; i++) {
        int min_idx = i;
        
        for (int j = i + 1; j < n; j++) {
            if (arr[j] < arr[min_idx]) {
                min_idx = j;
            }
        }
        
        // Swap the found minimum element with the element at index i
        if (min_idx != i) {
            swap(arr[i], arr[min_idx]);
        }
    }
}

Time Complexity:

  • Best Case: O(n)
  • Average Case: O(n)
  • Worst Case: O(n)
  • Space Complexity: O(1)

Insertion Sort

void insertionSort(vector<int>& arr) {
    int n = arr.size();
    
    for (int i = 1; i < n; i++) {
        int key = arr[i];
        int j = i - 1;
        
        // Move elements greater than key to one position ahead
        while (j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j];
            j--;
        }
        arr[j + 1] = key;
    }
}

Time Complexity:

  • Best Case: O(n) when already sorted
  • Average Case: O(n)
  • Worst Case: O(n)
  • Space Complexity: O(1)

Merge Sort

void merge(vector<int>& arr, int left, int mid, int right) {
    int n1 = mid - left + 1;
    int n2 = right - mid;
    
    // Create temporary arrays
    vector<int> L(n1), R(n2);
    
    // Copy data to temporary arrays
    for (int i = 0; i < n1; i++)
        L[i] = arr[left + i];
    for (int i = 0; i < n2; i++)
        R[i] = arr[mid + 1 + i];
    
    // Merge the temporary arrays back into arr[left..right]
    int i = 0, j = 0, k = left;
    
    while (i < n1 && j < n2) {
        if (L[i] <= R[j]) {
            arr[k] = L[i];
            i++;
        } else {
            arr[k] = R[j];
            j++;
        }
        k++;
    }
    
    // Copy the remaining elements of L[], if any
    while (i < n1) {
        arr[k] = L[i];
        i++;
        k++;
    }
    
    // Copy the remaining elements of R[], if any
    while (j < n2) {
        arr[k] = R[j];
        j++;
        k++;
    }
}

void mergeSort(vector<int>& arr, int left, int right) {
    if (left < right) {
        int mid = left + (right - left) / 2;
        
        // Sort first and second halves
        mergeSort(arr, left, mid);
        mergeSort(arr, mid + 1, right);
        
        // Merge the sorted halves
        merge(arr, left, mid, right);
    }
}

// Wrapper function
void mergeSort(vector<int>& arr) {
    mergeSort(arr, 0, arr.size() - 1);
}

Time Complexity:

  • Best Case: O(n log n)
  • Average Case: O(n log n)
  • Worst Case: O(n log n)
  • Space Complexity: O(n)

Quick Sort

int partition(vector<int>& arr, int low, int high) {
    int pivot = arr[high];  // Choose rightmost element as pivot
    int i = low - 1;  // Index of smaller element
    
    for (int j = low; j < high; j++) {
        // If current element is smaller than the pivot
        if (arr[j] < pivot) {
            i++;
            swap(arr[i], arr[j]);
        }
    }
    
    // Place pivot in its correct position
    swap(arr[i + 1], arr[high]);
    return i + 1;
}

void quickSort(vector<int>& arr, int low, int high) {
    if (low < high) {
        // pi is partitioning index, arr[pi] is now at right place
        int pi = partition(arr, low, high);
        
        // Separately sort elements before and after partition
        quickSort(arr, low, pi - 1);
        quickSort(arr, pi + 1, high);
    }
}

// Wrapper function
void quickSort(vector<int>& arr) {
    quickSort(arr, 0, arr.size() - 1);
}

Time Complexity:

  • Best Case: O(n log n)
  • Average Case: O(n log n)
  • Worst Case: O(n) (when already sorted and rightmost element chosen as pivot)
  • Space Complexity: O(log n) for recursion stack

Heap Sort

void heapify(vector<int>& arr, int n, int i) {
    int largest = i;  // Initialize largest as root
    int left = 2 * i + 1;
    int right = 2 * i + 2;
    
    // If left child is larger than root
    if (left < n && arr[left] > arr[largest])
        largest = left;
    
    // If right child is larger than largest so far
    if (right < n && arr[right] > arr[largest])
        largest = right;
    
    // If largest is not root
    if (largest != i) {
        swap(arr[i], arr[largest]);
        
        // Recursively heapify the affected sub-tree
        heapify(arr, n, largest);
    }
}

void heapSort(vector<int>& arr) {
    int n = arr.size();
    
    // Build max heap
    for (int i = n / 2 - 1; i >= 0; i--)
        heapify(arr, n, i);
    
    // Extract elements from heap one by one
    for (int i = n - 1; i > 0; i--) {
        // Move current root to end
        swap(arr[0], arr[i]);
        
        // Call max heapify on the reduced heap
        heapify(arr, i, 0);
    }
}

Time Complexity:

  • Best Case: O(n log n)
  • Average Case: O(n log n)
  • Worst Case: O(n log n)
  • Space Complexity: O(1)

Binary Search

// Iterative implementation
int binarySearch(const vector<int>& arr, int target) {
    int left = 0;
    int right = arr.size() - 1;
    
    while (left <= right) {
        int mid = left + (right - left) / 2;
        
        // Check if target is present at mid
        if (arr[mid] == target) {
            return mid;
        }
        
        // If target is greater, ignore left half
        if (arr[mid] < target) {
            left = mid + 1;
        }
        // If target is smaller, ignore right half
        else {
            right = mid - 1;
        }
    }
    
    // Element not present
    return -1;
}

// Recursive implementation
int binarySearchRecursive(const vector<int>& arr, int left, int right, int target) {
    if (right >= left) {
        int mid = left + (right - left) / 2;
        
        // If the element is present at the middle
        if (arr[mid] == target) {
            return mid;
        }
        
        // If element is smaller than mid, search in left subarray
        if (arr[mid] > target) {
            return binarySearchRecursive(arr, left, mid - 1, target);
        }
        
        // Else search in right subarray
        return binarySearchRecursive(arr, mid + 1, right, target);
    }
    
    // Element not present
    return -1;
}

// Wrapper function for recursive binary search
int binarySearchRecursive(const vector<int>& arr, int target) {
    return binarySearchRecursive(arr, 0, arr.size() - 1, target);
}

Time Complexity:

  • Best Case: O(1) (when the target is the middle element)
  • Average Case: O(log n)
  • Worst Case: O(log n)
  • Space Complexity: O(1) iterative, O(log n) recursive

// DP approach with tabulation (bottom-up)
int fibTabulation(int n) {
    if (n <= 1) return n;
    
    vector<int> dp(n + 1);
    dp[0] = 0;
    dp[1] = 1;
    
    for (int i = 2; i <= n; i++) {
        dp[i] = dp[i - 1] + dp[i - 2];
    }
    
    return dp[n];
}

        a = b;
        b = c;

Longest Increasing Subsequence (LIS)

int longestIncreasingSubsequence(const vector<int>& nums) {
    int n = nums.size();
    vector<int> dp(n, 1);  // dp[i] = LIS ending at index i
    
    for (int i = 1; i < n; i++) {
        for (int j = 0; j < i; j++) {
            if (nums[i] > nums[j]) {
                dp[i] = max(dp[i], dp[j] + 1);
            }
        }
    }
    
    return *max_element(dp.begin(), dp.end());
}
// Time Complexity: O(n)
// Space Complexity: O(n)

int longestIncreasingSubsequenceOptimized(const vector<int>& nums) {

0/1 Knapsack Problem

int knapsack(const vector<int>& weights, const vector<int>& values, int W) {
    int n = weights.size();
    vector<vector<int>> dp(n + 1, vector<int>(W + 1, 0));
    
    for (int i = 1; i <= n; i++) {
        for (int w = 1; w <= W; w++) {
            // If current item weight is less than capacity, we have a choice
            if (weights[i - 1] <= w) {
                // Max of including or excluding current item
                dp[i][w] = max(values[i - 1] + dp[i - 1][w - weights[i - 1]], dp[i - 1][w]);
            } else {
                // Can't include current item
                dp[i][w] = dp[i - 1][w];
            }
        }
    }
    
    return dp[n][W];
}
// Time Complexity: O(n * W)
// Space Complexity: O(n * W)

// Space-optimized solution
int knapsackOptimized(const vector<int>& weights, const vector<int>& values, int W) {
    int n = weights.size();
    vector<int> dp(W + 1, 0);
    
    for (int i = 0; i < n; i++) {
        for (int w = W; w >= weights[i]; w--) {
            dp[w] = max(dp[w], values[i] + dp[w - weights[i]]);
        }
    }
    
    return dp[W];
}
// Time Complexity: O(n * W)
// Space Complexity: O(W)

Activity Selection

int maxActivities(vector<pair<int, int>>& activities) {
    // Sort activities by finish time
    sort(activities.begin(), activities.end(), 
         [](const pair<int, int>& a, const pair<int, int>& b) {
             return a.second < b.second;
         });
    
    int count = 1;  // First activity is always selected
    int lastFinish = activities[0].second;
    
    for (int i = 1; i < activities.size(); i++) {
        // If current activity start time is >= last finish time
        if (activities[i].first >= lastFinish) {
            count++;
            lastFinish = activities[i].second;
        }
    }
    
    return count;
}
// Time Complexity: O(n log n) due to sorting
// Space Complexity: O(1)

Fractional Knapsack

double fractionalKnapsack(vector<pair<int, int>>& items, int capacity) {
    // items: (value, weight)
    // Sort by value/weight ratio (decreasing)
    sort(items.begin(), items.end(), 
         [](const pair<int, int>& a, const pair<int, int>& b) {
             return (double)a.first / a.second > (double)b.first / b.second;
         });
    
    double totalValue = 0.0;
    
    for (auto& item : items) {
        if (capacity >= item.second) {
            // Take the whole item
            totalValue += item.first;
            capacity -= item.second;
        } else {
            // Take a fraction of the item
            totalValue += (double)item.first * capacity / item.second;
            break;
        }
    }
    
    return totalValue;
}
// Time Complexity: O(n log n) due to sorting
// Space Complexity: O(1)

Backtracking

N-Queens Problem

bool isSafe(vector<string>& board, int row, int col, int n) {
    // Check row
    for (int i = 0; i < col; i++) {
        if (board[row][i] == 'Q') return false;
    }
    
    // Check upper diagonal
    for (int i = row, j = col; i >= 0 && j >= 0; i--, j--) {
        if (board[i][j] == 'Q') return false;
    }
    
    // Check lower diagonal
    for (int i = row, j = col; i < n && j >= 0; i++, j--) {
        if (board[i][j] == 'Q') return false;
    }
    
    return true;
}

bool solveNQueensUtil(vector<string>& board, int col, int n) {
    // Base case: If all queens are placed
    if (col >= n) return true;
    
    // Try placing queen in all rows of current column
    for (int i = 0; i < n; i++) {
        // Check if queen can be placed
        if (isSafe(board, i, col, n)) {
            // Place the queen
            board[i][col] = 'Q';
            
            // Recur to place rest of the queens
            if (solveNQueensUtil(board, col + 1, n)) {
                return true;
            }
            
            // Backtrack if placing queen doesn't lead to a solution
            board[i][col] = '.';
        }
    }
    
    // If no row works, return false
    return false;
}

vector<string> solveNQueens(int n) {
    vector<string> board(n, string(n, '.'));
    
    if (!solveNQueensUtil(board, 0, n)) {
        // No solution exists
        return {};
    }
    
    return board;
}
// Time Complexity: O(n!)
// Space Complexity: O(n)

Sudoku Solver

bool isSafe(vector<vector<char>>& board, int row, int col, char num) {
    // Check row
    for (int x = 0; x < 9; x++) {
        if (board[row][x] == num) return false;
    }
    
    // Check column
    for (int x = 0; x < 9; x++) {
        if (board[x][col] == num) return false;
    }
    
    // Check 3x3 box
    int startRow = row - row % 3;
    int startCol = col - col % 3;
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            if (board[i + startRow][j + startCol] == num) return false;
        }
    }
    
    return true;
}

bool solveSudoku(vector<vector<char>>& board) {
    int row, col;
    bool isEmpty = false;
    
    // Find empty cell
    for (row = 0; row < 9; row++) {
        for (col = 0; col < 9; col++) {
            if (board[row][col] == '.') {
                isEmpty = true;
                break;
            }
        }
        if (isEmpty) break;
    }
    
    // No empty cell found, puzzle solved
    if (!isEmpty) return true;
    
    // Try digits 1-9 for empty cell
    for (char num = '1'; num <= '9'; num++) {
        if (isSafe(board, row, col, num)) {
            // Make tentative assignment
            board[row][col] = num;
            
            // Return if success
            if (solveSudoku(board)) return true;
            
            // Failure, undo assignment
            board[row][col] = '.';
        }
    }
    
    // Trigger backtracking
    return false;
}
// Time Complexity: O(9^(n))
// Space Complexity: O(n)

Common Patterns

Two Pointers

// Example: Two Sum (find indices of two numbers that add up to target)
vector<int> twoSum(vector<int>& nums, int target) {
    // Store copy of original array with indices
    vector<pair<int, int>> numWithIndex;
    for (int i = 0; i < nums.size(); i++) {
        numWithIndex.push_back({nums[i], i});
    }
    
    // Sort the array
    sort(numWithIndex.begin(), numWithIndex.end());
    
    // Two pointers approach
    int left = 0;
    int right = numWithIndex.size() - 1;
    
    while (left < right) {
        int sum = numWithIndex[left].first + numWithIndex[right].first;
        
        if (sum == target) {
            return {numWithIndex[left].second, numWithIndex[right].second};
        } else if (sum < target) {
            left++;
        } else {
            right--;
        }
    }
    
    return {-1, -1};  // No solution found
}
// Time Complexity: O(n log n)
// Space Complexity: O(n)

Sliding Window

// Example: Maximum sum of a subarray of size k
int maxSubarraySum(const vector<int>& arr, int k) {
    int n = arr.size();
    
    // Handle edge cases
    if (n < k) return -1;
    
    // Compute sum of first window
    int maxSum = 0;
    for (int i = 0; i < k; i++) {
        maxSum += arr[i];
    }
    
    // Compute sums of remaining windows by sliding
    int windowSum = maxSum;
    for (int i = k; i < n; i++) {
        windowSum += arr[i] - arr[i - k];
        maxSum = max(maxSum, windowSum);
    }
    
    return maxSum;
}
// Time Complexity: O(n)
// Space Complexity: O(1)

Fast and Slow Pointers

// Example: Detect cycle in a linked list
bool hasCycle(ListNode* head) {
    if (!head || !head->next) return false;
    
    ListNode* slow = head;
    ListNode* fast = head;
    
    while (fast && fast->next) {
        slow = slow->next;
        fast = fast->next->next;
        
        if (slow == fast) return true;  // Cycle detected
    }
    
    return false;  // No cycle
}
// Time Complexity: O(n)
// Space Complexity: O(1)

// Find leftmost occurrence of target
int lowerBound(const vector<int>& nums, int target) {
    int left = 0;
    int right = nums.size();
    
    while (left < right) {
        int mid = left + (right - left) / 2;
        
        if (nums[mid] < target) {
            left = mid + 1;
        } else {
            right = mid;
        }

}

        if (nums[mid] <= target) {
            left = mid + 1;
        } else {
            right = mid;
        }
    }
    
    return left - 1;
}

  2. Develop a strategy:

    • Start with a brute force solution
    • Identify inefficiencies and optimize
    • Consider common patterns (two pointers, sliding window, etc.)

  3. Code the solution:

    • Write clean, modular code
    • Use meaningful variable names
    • Add comments for complex logic

  4. Test your solution:

    • Test with normal cases
    • Test edge cases (empty, single element, etc.)
    • Trace through with a small example

  5. Analyze complexity:

    • Discuss time and space complexity
    • Propose further optimizations if possible

Common Mistakes to Avoid

  • Not considering edge cases
  • Overcomplicated solutions
  • Forgetting to free allocated memory
  • Off-by-one errors
  • Integer overflow/underflow
  • Not handling error conditions
  • Mixing 0-indexed and 1-indexed thinking

C++ Specific Tips

  • Be careful with pass-by-value vs. pass-by-reference
  • Use const for parameters that shouldnt be modified
  • Avoid unnecessary dynamic memory allocation
  • Know when to use different STL containers
  • Watch out for iterator invalidation
  • Remember to check for null pointers
  • Understand move semantics and perfect forwarding (modern C++)

Communication

  • Think out loud during problem-solving
  • Explain your thought process
  • Discuss trade-offs between different approaches
  • Be receptive to hints and feedback
  • Ask for clarification when needed

Remember, interviews assess not just your technical skills but also your problem-solving approach and communication abilities. Practice regularly and focus on understanding the fundamental concepts rather than memorizing solutions.
